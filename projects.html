7) Write a C# program to check if a number entered by user, is even or odd.

Sample Input: 15

Expected Output: Number 15 is an odd number.

using System;
namespace EvenOrOdd
{
    public class MyProgram
    {
        public static void Main()
        {
            int x;
            Console.Write("Enter a number: ");
            x=Convert.ToInt32(Console.ReadLine());
            if(x%2==0){
                Console.Write($"Number {x} is an even number.");
            }
            else{
                Console.Write($"Number {x} is an odd number.");
            }
        }
    }
}
